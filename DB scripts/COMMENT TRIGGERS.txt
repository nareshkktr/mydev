DELIMITER //
CREATE TRIGGER UPDATE_REPLY_COUNT_INSERT AFTER INSERT ON COMMENT_MASTER
FOR EACH ROW
BEGIN
	IF NEW.PARENT_COMMENT_ID IS NOT NULL THEN
		UPDATE COMMENT_MASTER SET REPLY_COUNT=REPLY_COUNT+1 WHERE COMMENT_GUID = NEW.PARENT_COMMENT_ID;
	END IF;
END//

DELIMITER ;

DELIMITER //
CREATE TRIGGER UPDATE_REPLY_COUNT_DELETE AFTER DELETE ON COMMENT_MASTER
FOR EACH ROW
BEGIN
	IF OLD.PARENT_COMMENT_ID IS NOT NULL THEN
		UPDATE COMMENT_MASTER SET REPLY_COUNT=REPLY_COUNT-1 WHERE COMMENT_GUID = OLD.PARENT_COMMENT_ID;
	END IF;
END//

DELIMITER ;
